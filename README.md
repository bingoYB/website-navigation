目录结构
---

```
├── README.md
├── package.json
├── action                # gitaction相关脚本
├── docs                  # 打包目录 实际页面访问目录
├── public                # 静态资源
├── src                   # 页面所有代码
│   ├── data              # 数据文件
│   ├── page              # 页面代码
│   ├── wget              # 页面零部件
│   ├── style             # 样式字体等
│   ├── assets            # 样式资源
│   ├── store             # redux全局状态
│   ├── router            # 路由组件
│   ├── main.jsx        # 页面入口文件
```
